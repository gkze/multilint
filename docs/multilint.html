multilint

Multilint is a runner for multiple code quality tools.

Multilint enables running various Python linters (and other CQ tools) under the same interface, making it easier to configure and run all code quality tools from a centralized location.

 69class ToolLogger(Logger):
 70    """ToolLogger allows setting format on itself during instantiation."""
 71
 72    def __init__(
 73        self: ToolLogger, name: str, level: LogLevel, logfmt: str = LOG_FMT
 74    ) -> None:
 75        """Create a ToolLogger with the specified name, level, and format.
 76
 77        Log format gets applied immediately.
 78        """
 79        super().__init__(name, level=level)
 80
 81        self.set_format(logfmt)
 82
 83    def set_format(self: ToolLogger, fmtstr: str) -> None:
 84        """Set the specified log message format.
 85
 86        Uses a StreamHandler by setting a formatter on it with the specified
 87        format string, and adds the StreamHanlder to the logger instance.
 88        """
 89        handler: StreamHandler = StreamHandler()
 90        handler.setFormatter(Formatter(fmtstr))
 91        self.addHandler(handler)

 94class TextIOLogger(TextIOBase, ToolLogger):
 95    """Logger object that can be written to like a stream-like object.
 96
 97    A logger that masquerades as a TextIO-compatible object, allowing it to be
 98    passed into code quality tools that write to TextIO interfaces. This way,
 99    it is possible to wrap the stdout / stderr / other streams with our
100    common logging.
101    """
102
103    def __init__(
104        self: TextIOLogger, name: str, level: LogLevel, logfmt: str = LOG_FMT
105    ) -> None:
106        """Construct a TextIOLogger."""
107        self._name: str = name
108        self._level: int = level
109
110        super().__init__(self._name, self._level, logfmt)
111
112    def write(self: TextIOLogger, msg: str) -> int:
113        """Write data to the logger as if it were a stream-like object.
114
115        The write() method is implemented to forward data to the logging
116        part of this object, allowing capturing logs that would normally be
117        written to stdout / stderr.
118        """
119        if msg in ("", "\n"):
120            return 0
121
122        for line in filter(None, msg.split("\n")):
123            self.log(self._level, line)
124
125        return len(msg)

128class ToolRunner:
129    """Base class for integrating code quality tools.
130
131    ToolRunner is a base class for any plugin that integrates a code quality
132    tool. Subclasses only have to implement the run() method. There is a
133    convenience method available,
134    """
135
136    def __init__(
137        self: ToolRunner,
138        tool: Tool,
139        src_paths: Seq[Path] = [Path(".")],
140        config: Mapping[str, Any] = {},
141    ) -> None:
142        """Initialize a ToolRunner object."""
143        self._tool: Tool = tool
144        self.src_paths: Seq[Path] = src_paths
145        self.config: Mapping[str, Any] = config
146
147    def make_logger(
148        self: ToolRunner,
149        cls: type[ToolLogger] | type[TextIOLogger],
150        level: LogLevel,
151    ) -> ToolLogger | TextIOLogger:
152        """Create a logger for the ToolRunner object.
153
154        Creates a logger object from the specified logger class (can be
155        either a ToolLogger or a TextIOLogger) with the specified default log
156        level.
157        """
158        return cls(f"tool.{self._tool.value}", level)
159
160    def run(self: ToolRunner) -> ToolResult:
161        """Is implemented by subclasses to run the CQ (code quality) tool."""
162        raise NotImplementedError("run() needs to be implemented by subclass!")

165class AutoflakeRunner(ToolRunner):
166    """Runs autoflake.
167
168    Autoflake removes unused imports and variables among other
169    things. Reads autoflake arguments from pyproject.toml. Arguments are
170    specified by their full name, with underscores or dashes - either style is
171    accepted.
172    """
173
174    def _make_autoflake_args(self: AutoflakeRunner) -> list[str]:
175        args: list[str] = []
176
177        for key, val in self.config.items():
178            if key == "src_paths":
179                for src in val:
180                    args.append(src)
181
182                continue
183
184            opt: str = f"--{key.replace('_', '-')}"
185
186            if isinstance(val, bool):
187                args.append(opt)
188
189                continue
190
191            args.append(f"{opt}={val}")
192
193        return args
194
195    def run(self: AutoflakeRunner) -> ToolResult:
196        """Run autoflake."""
197        logger: Logger = self.make_logger(TextIOLogger, logging.INFO)
198        autoflake_args: list[str] = self._make_autoflake_args()
199        if all(cfgval.startswith("--") for cfgval in autoflake_args):
200            autoflake_args.extend([str(p) for p in self.src_paths])
201
202        retcode: int = autoflake_main(
203            [self._tool.value, *autoflake_args], logger, logger
204        )
205
206        return ToolResult.SUCCESS if retcode == 0 else ToolResult.FAILURE

226class ISortRunner(ToolRunner):
227    """Runs isort.
228
229    Isort is able to sort imports in a Python source file according to a defined
230    rule set (with sensible defaults).
231    """
232
233    # pylint: disable=too-many-branches
234    def run(self: ISortRunner) -> ToolResult:
235        """Run isort."""
236        logger: Logger = self.make_logger(ToolLogger, logging.INFO)
237        results: set[ISortResult] = set()
238
239        isort_logger: TextIOLogger = cast(
240            TextIOLogger, self.make_logger(TextIOLogger, logging.INFO)
241        )
242
243        # fmt: off
244        for file in isort_files.find(
245            [str(p) for p in self.src_paths]
246            or cast(Iterable[str], DEFAULT_CONFIG.src_paths),
247            DEFAULT_CONFIG, [], [],
248        ):
249            try:
250                with patch("sys.stdout", isort_logger):
251                    isort_file(file)
252
253                results.add(ISortResult(Path(file), ToolResult.SUCCESS))
254            # pylint: disable=broad-except
255            except Exception as ex:
256                results.add(ISortResult(
257                    Path(file), ToolResult.FAILURE,
258                    getattr(ex, "message") if hasattr(ex, "message") else "",
259                ))
260        # fmt: on
261
262        failed: set[ISortResult] = set()
263        for isort_result in results:
264            if isort_result.result == ToolResult.FAILURE:
265                failed.add(isort_result)
266
267        if len(failed) > 0:
268            logger.error("isort failed on some files:")
269
270            for failed_result in failed:
271                logger.error(f"{failed_result.pyfile}: {failed_result.errmsg}")
272
273        if len(failed) == len(results):
274            return ToolResult.FAILURE
275
276        if 0 < len(failed) < len(results):
277            return ToolResult.SUCCESS_PARTIAL
278
279        return ToolResult.SUCCESS

388class PyupgradeRunner(ToolRunner):
389    """Runs Pyupgrade.
390
391    Pyupgrade automatically upgrades Python syntax to the latest for the
392    specified Python version.
393    """
394
395    def _validate_config(self: PyupgradeRunner) -> None:
396        if "min_version" in self.config and not re.match(
397            r"^[0-9].[0-9]", self.config["min_version"]
398        ):
399            raise ValueError("min_version must be a valid Python version!")
400
401    def run(self: PyupgradeRunner) -> ToolResult:
402        """Run Pyupgrade."""
403        self._validate_config()
404
405        logger: ToolLogger = self.make_logger(TextIOLogger, logging.INFO)
406
407        with patch("sys.stdout", logger), patch("sys.stderr", logger):
408            retcode: int = 0
409            for src_path in self.src_paths:
410                retcode |= pyupgrade_fix_file(
411                    src_path,
412                    Namespace(
413                        exit_zero_even_if_changed=self.config.get(
414                            "exit_zero_even_if_changed", None
415                        ),
416                        keep_mock=self.config.get("keep_mock", None),
417                        keep_percent_format=self.config.get(
418                            "keep_percent_format", None
419                        ),
420                        keep_runtime_typing=self.config.get(
421                            "keep_runtime_typing", None
422                        ),
423                        min_version=tuple(
424                            int(v)
425                            for v in cast(
426                                str, self.config.get("min_version", "2.7")
427                            ).split(".")
428                        ),
429                    ),
430                )
431
432            return ToolResult.SUCCESS if retcode == 0 else ToolResult.FAILURE

435def find_pyproject_toml(path: Path = Path(".")) -> Path | None:
436    """Discover closest pyproject.toml.
437
438    Finds the first pyproject.toml by searching in the current directory,
439    traversing upward to the filesystem root if not found.
440    """
441    if path == Path("/"):
442        return None
443
444    filepath: Path = path / PYPROJECT_TOML_FILENAME
445
446    if filepath.exists() and filepath.is_file():
447        return path / PYPROJECT_TOML_FILENAME
448
449    return find_pyproject_toml(path.resolve().parent)

452def parse_pyproject_toml(pyproject_toml_path: Path = Path(".")) -> Mapping[str, Any]:
453    """Parse pyproject.toml into a config map.
454
455    Reads in the pyproject.toml file and returns a parsed version of it as a
456    Mapping.
457    """
458    pyproject_toml: Path | None = find_pyproject_toml(pyproject_toml_path)
459    if pyproject_toml is None:
460        return {}
461
462    with pyproject_toml.open("r") as file:
463        return toml.load(file)

466def expand_src_paths(src_paths: Seq[Path]) -> list[Path]:
467    """Expand source paths in case they are globs."""
468    return sum(
469        (
470            [Path(ge) for ge in glob(p.name)] if "*" in p.name else [p]
471            for p in src_paths
472        ),
473        [],
474    )

488class Multilint:
489    """The core logic of this project.
490
491    Multilint ties together multiple linting and other code quality tools
492    under a single interface, the ToolRunner base class. By subclassing
493    ToolRunner and implementing its .run() method, adding support for linters /
494    other tool plugins is easy.
495    """
496
497    DEFAULT_TOOL_ORDER: Seq[Tool] = [
498        Tool.PYUPGRADE,
499        Tool.AUTOFLAKE,
500        Tool.ISORT,
501        Tool.BLACK,
502        Tool.MYPY,
503        Tool.PYLINT,
504        Tool.PYDOCSTYLE,
505    ]
506
507    def __init__(
508        self: Multilint,
509        src_paths: Seq[Path] = [Path(".")],
510        pyproject_toml_path: Path = Path(".") / PYPROJECT_TOML_FILENAME,
511    ) -> None:
512        """Construct a new Multilint instance."""
513        self._config: Mapping[str, Any] = {}
514        self._config = parse_pyproject_toml(pyproject_toml_path)
515
516        self._multilint_config = self._get_tool_config(Tool.MULTILINT)
517        self._src_paths: Seq[Path] = (
518            src_paths
519            if len(src_paths) > 0
520            else self._multilint_config.get("src_paths", ["."])
521        )
522
523        self._tool_order: Seq[Tool] = [
524            Tool(t) for t in self._multilint_config.get("tool_order", [])
525        ]
526        if len(self._tool_order) == 0:
527            self._tool_order = self.DEFAULT_TOOL_ORDER
528
529    def _get_tool_config(self: Multilint, tool: Tool) -> Mapping[str, Any]:
530        return self._config.get("tool", {}).get(tool.value, {})
531
532    def run_tool(self: Multilint, tool: Tool) -> ToolResult:
533        """Run a single CQ tool.
534
535        Runs a single specified linter or other code quality tool. Returns
536        a ToolResult from the run.
537        """
538        LOGGER.info(f"Running {tool.value}...")
539        tool_config: Mapping[str, Any] = self._get_tool_config(tool)
540
541        # fmt: off
542        result: ToolResult = cast(ToolRunner, TOOL_RUNNERS[tool](
543            tool,
544            expand_src_paths(
545                [Path(sp) for sp in tool_config.get("src_paths", self._src_paths)]
546            ),
547            tool_config,
548        )).run()
549        # fmt: on
550
551        LOGGER.info(f"{tool.value} exited with {result}")
552
553        return result
554
555    def run_all_tools(
556        self: Multilint, order: Seq[Tool] = None
557    ) -> Mapping[Tool, ToolResult]:
558        """Run tools in specified order."""
559        results: dict[Tool, ToolResult] = {}
560
561        if order is None:
562            order = self._tool_order
563
564        for tool in order:
565            results[tool] = self.run_tool(tool)
566
567        return results
